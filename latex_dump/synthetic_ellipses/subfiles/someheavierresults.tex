
\section{Some Heavier results}

\begin{theorem}[Carnot's theorem]
Let there be 6 points on a triangle $\Delta ABC$, two on each side. Then there exists a conic passing through those 6 points if and only if
$${\frac {|AC_{A}|}{|BC_{A}|}}\cdot {\frac {|AC_{B}|}{|BC_{B}|}}\cdot {\frac {|BA_{B}|}{|CA_{B}|}}\cdot {\frac {|BA_{C}|}{|CA_{C}|}}\cdot {\frac {|CB_{C}|}{|AB_{C}|}}\cdot {\frac {|CB_{A}|}{|AB_{A}|}}=1$$
\end{theorem}
